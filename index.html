滚轮事件的应用 

<style type="text/css">
#box{
    width:300px;
    height:300px;
    background-color:aliceblue;
}
</style>
<script typr="text/javascript">
window.onload = function(){
    var box = document.getElementById("box");
    box.onwheel = function(event){            //onwheel支持所有浏览器
        if(event.wheelDelta>0){
        box.style.height = box.clientHeight - 10 +"px";           //.wheelDelta返回的正负值为滚轮上下方向
        }else{
        box.style.height = box.clientHeight + 10 +"px";};
    }//往上下滚轮改变对象块的高度
}
//如页面过大会导致调用浏览器默认滚动条 屏蔽浏览器默认行为返回false取消
</script>

对象的拖拽与产生问题的解决 

<style type="text/css">
#box{
    width:300px;
    height:300px;
    background-color:aliceblue;
    position:absolute;
}
    #box2{
    width:300px;
    height:300px;
    background-color:#bfa;
    position:absolute;
    left:500px;
    top:500px;
}

</style>
<script type="text/javascript">
window.onload = function(){
    var box = document.getElementById("box");
    box.onmousedown = function(event){
    var ofX = event.clientX - box.offsetLeft;
    var ofY = event.clientY - box.offsetTop;   //使对象可以任意位置拖拽不突变
        document.onmousemove = function(event){
        var left = event.clientX - ofX;
        var top = event.clientY - ofY;
        box.style.left = left+"px";
        box.style.top = top+"px";
        }
        document.onmouseup = function(){  //如果事件帮给box 会导致兄弟元素重合无法松开拖拽 应当给document绑定！
        document.onmousemove = null;      //up写在down里
        document.onmouseup = null;        //防止其他地方松开鼠标时触发 关闭函数
        }
    return false;}//防止全选时拖拽时适用浏览器默认事件

}//此方法滚动条时会错位
1</script>

事件的冒泡及子元素的取消冒泡 

<style type="text/css">
#box1{
    width:300px;
    height:300px;
    background-color:aliceblue;
}
#sp1{
    background-color:#bfa;
}
</style>
<script type="text/javascript">
window.onload = function(){
var sp1 = document.getElementById("sp1");//参数带引号！
var box1 = document.getElementById("box1");

sp1.onclick = function(event){
   
    alert("spanalt");
    event.cancelBubble = true;  //此函数取消冒泡 点击产生一次告警
    }

box1.onclick = function(){
    alert("divalt");    //未取消冒泡 产生2次告警
    }
document.body.onclick = function(){
    alert("bodyalt");
    }

}

</script>

<style type="text/css">
#article{
    width:500px;
    height:800px;
    background-color:aliceblue;
    overflow:auto;
    font-size:20px;
}
</style>
<script>
window.onload = function(){
//onscroll 在滚动条滚动时触发的事件
var article = document.getElementById("article");
var input = document.getElementsByTagName("input");  //获得的是类Array
article.onscroll = function(){
    if(article.scrollHeight - article.scrollTop == article.clientHeight){   //滚动块实际块总高度-滚动距离==块定义高度/可见高度
    input[0].disabled=false;                    //disabled为bool变量 true为禁用
   }

};
input[0].onclick = function(){
if(input[0].checked){
          input[1].disabled=false;
}else{input[1].disabled=true;}
}
