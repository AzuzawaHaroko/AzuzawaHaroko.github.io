注意数组长度比索引要+1

function waiFu(name,age){
this.name = name;
this.age = age;

};
var kyoka = new waiFu("镜华",8);
var mimi = new waiFu("美美",9);
var bomber = new waiFu("炸弹人",10);
var xcw = new waiFu("小仓唯",26);
var crn = new waiFu("日高里菜",25);

var waiFuArr = [kyoka,mimi,bomber,xcw,crn];


function girudo(shuzu){        //遍历选出小小甜心成员 （age<=10）
var xxtx=[];
    for(i=0 ; i < shuzu.length ; i++)
        {if(shuzu[i].age <= 10)
        xxtx.push(shuzu[i]);
                        
        };
        
        return xxtx;

};

console.log(girudo(waiFuArr));
