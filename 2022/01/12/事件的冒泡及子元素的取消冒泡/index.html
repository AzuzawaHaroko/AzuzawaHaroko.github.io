事件的冒泡及子元素的取消冒泡 

<style type="text/css">
#box1{
    width:300px;
    height:300px;
    background-color:aliceblue;
}
#sp1{
    background-color:#bfa;
}
</style>
<script type="text/javascript">
window.onload = function(){
var sp1 = document.getElementById("sp1");//参数带引号！
var box1 = document.getElementById("box1");

sp1.onclick = function(event){
   
    alert("spanalt");
    event.cancelBubble = true;  //此函数取消冒泡 点击产生一次告警
    }

box1.onclick = function(){
    alert("divalt");    //未取消冒泡 产生2次告警
    }
document.body.onclick = function(){
    alert("bodyalt");
    }

}

</script>

//span为div的子元素
